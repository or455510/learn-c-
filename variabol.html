variables المتغيرات

المتغيرات بشكل مبسط هي مخازن أو حوامل للقيم مخزنة في الذاكرة بحيث كل موقع في الذاكرة له عنوان وقيمة خاصة

بالعنوان وكل متغير يحمل قيمة خاصة به وللمتغيرات أهمية لا يمكن الأستغناء عنها في العمليات البرمجية .

Syntax

Typevariable = value;

نوع_المتغير اسم_المتغير = قيمة_المتغير

أنواع المتغيرات

خاص بتحميل القيم الصحيحة فقط int النوع

int myNum = 15;

خاص بتحميل القيم العشرية فقط Doubleالنوع

double myFloatNum = 5.99;

خاص بتحميل القيم النصية Stringالنوع

string myText = "Hello";

أو غيرها من الحروف والرموز والارقام B او حرف Aيحمل رمز واحد فقط مثلا حرف Charالنوع

char myLetter = 'D';

وتعني صحيحة أو خاطئة true or false يحمل فقط قيمتين Booleanالنوع

bool myBoolean1 = true;

bool myBoolean2 = false;

طرق تهيئة المتغيرات

تعريف متغير واعطاء القيمة بشكل مباشر :

int myNum = 15;

تعريف متغير واعطاء القيمة بشكل غير مباشر :

int myNum ;

myNum = 15;

تعريف متغير واعطاء القيمة بشكل مباشر ومن ثم تغيير القيمة :

int myNum = 15; // myNum is 15

myNum = 10; // Now myNum is 10

الدمج بين القيم في المتغيرات

<< الدمج في الطباعة يتم الدمج بين النصوص والقيم الصحيحة باستخدام العلامة

#include <iostream>

using namespace std;

int main(){

string myName = "omar";

cout << "myName is :" <<myName <<endl;

int myAge = 25;

cout << "myAge is :" <<myAge <<endl;

return 0;

}

Output :

myName is :Ahmed

myAge is :25

الدمج في المتغيرات يتم الدمج بين بين المتغيرات باستخدام العلامة+ , وللتنبيه تستخدم العلامة+ للدمج اذا كانت المتغيرات نصوص وتستخدم للجمع اذا كانت قيم صحيحة

#include <iostream>

using namespace std;

int main(){

string firstName= "omar";

string lastName= "ramadan";

string fullName=firstName+lastName; ;

cout << fullName

return 0;

}

Output :

omar ramadan
