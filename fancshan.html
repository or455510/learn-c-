في هذا المثال تم أستخدام اربعة دوال لأجراء عمليات الجمع والطرح والضرب والقسمة

Void SumFunction(intnum1 ,intnum2 ){

Cout <<"sum" <<sum << endl;

}

Void SubFunction(intnum1 ,intnum2 ){

intSub=num1-num2

Cout <<"Sub" <<Sub << endl;

}

Void MulFunction(intnum1 ,intnum2 ){

intMul=num1*num2

Cout <<"Mul" <<Mul << endl;

}

Void DivFunction(intnum1 ,intnum2 ){

intDiv=num1/num2

Cout <<"Div" <<Div << endl;

}

int main(){

SumFunction(5,10);

SumFunction(10,20);

SumFunction(2,6);

SumFunction(10,2);

Return 0;

}

Output :

Sum = 15

Sub = 10

Mul = 12

Div = 5
